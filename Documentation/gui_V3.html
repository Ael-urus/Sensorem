 316    def est_nom_fichier_valide(self,nom: str) -> bool:
 317        """Vérifie si un nom est valide pour un fichier, en tenant compte des contraintes multi-OS.
 318
 319        Args:
 320            nom: Le nom à vérifier.
 321
 322        Returns:
 323            True si le nom est valide, False sinon.
 324        """
 325
 326        if not nom:  # Vérification pour les chaînes vides
 327            return False
 328
 329        if len(nom) > 255:  # Vérification de la longueur maximale
 330            return False
 331
 332        # Caractères interdits universellement (tous OS)
 333        interdits = r'[\\/:*?"<>|]'
 334        if re.search(interdits, nom):
 335            return False
 336
 337        # Caractères interdits sous Windows
 338        if os.name == 'nt':
 339            if re.search(r'[\x00-\x1F]', nom):  # Caractères de contrôle ASCII
 340                return False
 341            if nom.endswith('.') or nom.endswith(' '):  # Noms finissant par un point ou un espace
 342                return False
 343            if nom in (".", ".."):  # Noms "." ou ".."
 344                return False
 345            if re.search(r'CON|PRN|AUX|NUL|COM[0-9]|LPT[0-9]', nom, re.IGNORECASE):  # Noms réservés Windows
 346                return False
 347
 348        # Caractères interdits sous macOS (principalement le caractère ':')
 349        if os.name == 'posix' and os.uname().sysname == 'Darwin':
 350            if ':' in nom:
 351                return False
 352
 353        # Caractères interdits sous Linux (principalement le caractère '/')
 354        if os.name == 'posix' and os.uname().sysname == 'Linux':
 355            if '/' in nom:  # Déjà géré par la vérification universelle, mais ajouté pour plus de clarté
 356                return False
 357
 358        return True

 540def afficher_graphique(donnees: List[List], zone_graphique: ttk.Frame, titre: str = "un titre") -> None:
 541    """Affiche un graphique dans la zone graphique.
 542
 543    Efface le contenu précédent de la zone graphique, récupère les unités et le nom du capteur de référence,
 544    puis trace le graphique avec les données fournies.
 545
 546    Args:
 547        donnees: Liste de listes de données à tracer.
 548        zone_graphique: Zone graphique où tracer le graphique.
 549        titre: Titre du graphique.
 550    """
 551    # Détruit tous les widgets enfants de la zone graphique pour effacer le graphique précédent.
 552    for widget in zone_graphique.winfo_children():
 553        widget.destroy()
 554
 555    # Récupère les variables d'unité et de nom du capteur de référence depuis l'objet fg.
 556    champ_unit_capteurs: tk.StringVar = fg.get_state("champ_unit_capteurs")
 557    champ_unit_capteur_ref: tk.StringVar = fg.get_state("champ_unit_capteur_ref")
 558    champ_nom_capteur_ref: tk.StringVar = fg.get_state("champ_nom_capteur_ref")
 559
 560    # Récupère les valeurs des unités et du nom, en supprimant les espaces superflus.
 561    unit: str = champ_unit_capteurs.get().strip()
 562    unit_ref: str = champ_unit_capteur_ref.get().strip()
 563    nom_ref: str = champ_nom_capteur_ref.get().strip()
 564
 565    # Crée les légendes pour le graphique.
 566    Legende: List[str] = ['Capteur (s) passé (s)' + unit, 'Capteur_ref :' + nom_ref + ' ' + unit_ref]
 567
 568    # Définit les styles pour les lignes du graphique.
 569    styles = [
 570        {'linestyle': '-', 'marker': 'None', 'linewidth': '0.6'},
 571        {'linestyle': '-', 'marker': 'None', 'linewidth': '0.6'}
 572    ]
 573    # Définit les marges du graphique.
 574    marges = {'left': 0.08, 'right': 0.935, 'top': 0.92, 'bottom': 0.12}
 575
 576    # Trace le graphique en utilisant la fonction tracer_graphique de l'objet fg.
 577    fig = fg.tracer_graphique(donnees, Legende, titre,
 578                              xlabel="Nombre de points d'acquisition",
 579                              ylabels=["Capteur (s) " + unit, "Capteur ref " + unit_ref],
 580                              styles=styles,
 581                              marges=marges)
 582
 583    # Crée un canvas Tkinter pour afficher le graphique.
 584    canvas: FigureCanvasTkAgg = FigureCanvasTkAgg(fig, master=zone_graphique)
 585    # Dessine le graphique sur le canvas.
 586    canvas.draw()
 587    # Empaquète le canvas dans la zone graphique pour l'afficher.
 588    canvas.get_tk_widget().pack(fill=tk.BOTH, expand=True)
